from fastapi import APIRouter, Request
from datetime import datetime, timedelta
import sqlite3

router = APIRouter()

@router.post("/dt.get_time_invested_summary")
async def get_time_invested_summary(request: Request):
    body = await request.json()
    period = body.get("params", {}).get("period", "week")
    now = datetime.utcnow()
    start = now - timedelta(days=7 if period=="week" else 30)
    conn = sqlite3.connect("dt/db/activity.sqlite")
    cur = conn.cursor()
    cur.execute("""
        SELECT SUM(duration_minutes) FROM activity_log
        WHERE timestamp BETWEEN ? AND ?
          AND activity_type IN ('learning','project_work','reflection')
    """, (start.isoformat(), now.isoformat()))
    total = cur.fetchone()[0] or 0
    conn.close()
    avg = round(total / max(1, (now - start).days), 2)
    return {
        "jsonrpc": "2.0",
        "result": {
            "period": period,
            "total_minutes": total,
            "avg_minutes_per_day": avg,
            "start_date": start.strftime("%Y-%m-%d"),
            "end_date": now.strftime("%Y-%m-%d")
        }
    }
